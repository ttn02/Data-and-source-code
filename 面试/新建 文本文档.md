- rest参数（可以接收任意数量的参数，收集成数组）

  - ~~~js
    // 定义一个求和函数，使用rest参数接收任意数量的参数
    function sum(...numbers) {
      let total = 0;
      for (let num of numbers) {
        total += num;
      }
      return total;
    }
    
    // 调用sum函数，传入任意数量的参数
    console.log(sum(1, 2, 3)); // 输出：6
    console.log(sum(1, 2, 3, 4, 5)); // 输出：15
    ~~~
